! + tab - html template
link + tab - add stylesheet
.className + tab - create div with this class
.className * 3 - create three divs with this class
tagName + tab - create tag

nav>ul>li
<nav>
  <ul>
    <li></li>
  </ul>
</nav>

a{Click Me} - tag with content
<a href="">Click Me</a>

div.first-class.second-class - multiple classes
<div class="first-class second-class"></div>

div#content.test - id and class
<div id="content" class="test"></div>

List
li>3{Line number $}
<li>Line number 1</li>
<li>Line number 2</li>
<li>Line number 3</li>

<ul class="list-inline">
	<li class="list-inline-item">Line number 1</li>

	<li class="list-inline-item">Line number 3</li>
